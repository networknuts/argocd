### Installing ArgoCD - from Gateway / JumpServer (IP=192.168.1.8 / 10.0.0.199)

kubectl create namespace argocd
kubectl apply -n argocd --server-side --force-conflicts -f https://raw.githubusercontent.com/argoproj/argocd/stable/manifests/install.yaml

### Access ArgoCD UI using Workstation

#1 - SSH to JumpServer
kubectl port-forward svc/argocd-server -n argocd 8080:443 --address 0.0.0.0 &

#2 - Open another terminal on your workstation 
     Generate your ssh keys using "ssh-keygen"
ssh -L 8443:localhost:8080 alok@192.168.1.8 -i /home/alok/.ssh/id_edxxxx

#3 - Open the browser
http://localhost:8443
